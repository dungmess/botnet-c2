import subprocess
import sys

def install(package):
    subprocess.check_call([sys.executable, "-m", "pip", "install", package])

modules = ["requests"]

for m in modules:
    try:
        __import__(m)
        print(f"{m} đã có")
    except ImportError:
        print(f"Đang cài {m}...")
        install(m)

print("Xong!")
